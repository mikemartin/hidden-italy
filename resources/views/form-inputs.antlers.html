{{# ==================================================== BESPOKE SELECT — OPTGROUPS ==================================================== #}}
{{ partial:typography/h2 content="Bespoke select (optgroups)" }} {{ partial:typography/p_small content="When you need grouped options driven by Statamic collections — like the booking form's tour picker — the standard Peak select can't produce <optgroup>s. Hand-roll a select directly in the template using the same class soup, then wire it into the Alpine form state via x-model. Lives at booking/index.antlers.html today." color="text-muted" class="!mb-0" }}
{{ partial:typography/h4 content="Antlers" class="!mb-0" }} 
<div class="relative">
    <select
        id="tour"
        name="tour"
        x-model="form.tour"
        @change="form.validate('tour')"
        :class="{
            'border-error focus:border-error focus:ring-error/20': form.invalid('tour'),
            'border-text/20 focus:border-accent focus:ring-accent': !form.invalid('tour')
        }"
        class="block w-full appearance-none rounded-full bg-card pl-5 pr-12 py-3 text-base lg:text-lg text-foreground border focus:outline-none focus:ring-2 transition"
    >
        <option value="">Select a tour</option>
        <optgroup label="Guided tours">
            {{ collection:guided sort="name:asc" }}
                <option value="{{ slug }}">{{ name or title }}</option>
            {{ /collection:guided }}
        </optgroup>
    </select>
    <x-lucide-chevron-down class="absolute right-4 top-1/2 -translate-y-1/2 w-5 h-5 text-accent pointer-events-none" />
</div>

{{# ==================================================== HIDDEN ==================================================== #}}
{{ partial:typography/h2 content="Hidden" }} {{ partial:typography/p_small content="Two patterns. Static value: use Statamic's type: hidden with a blueprint default — the partial auto-renders <input type="hidden">. Dynamic per-request value (e.g. user_id from current_user): emit the hidden input manually in the template and seed the Alpine state via x-init="form.handle = $el.value" so Precognition picks it up. Don't fabricate values just to satisfy a validator — drop the field instead." color="text-muted" class="!mb-0" }}
{{ partial:typography/h4 content="Blueprint — static" class="!mb-0" }} 
- handle: source
  field:
    display: Source
    type: hidden
    default: 'booking-form'
{{ partial:typography/h4 content="Antlers — dynamic" class="!mb-0 mt-4" }} 
{{ current_user }}
    <input type="hidden" name="user_id" value="{{ id }}" x-init="form.user_id = $el.value">
{{ /current_user }}
